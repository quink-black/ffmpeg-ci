add_library(ffmpegplay SHARED

        ffmpeg.c

        ${ffmpeg_src_dir}/fftools/ffmpeg.h

        ${ffmpeg_src_dir}/fftools/sync_queue.c
        ${ffmpeg_src_dir}/fftools/thread_queue.c
